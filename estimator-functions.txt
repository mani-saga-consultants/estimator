public final enforceLsuLimits){
/**********  
<CFOUTPUT
	<br /> _lsuLimit = #getTuitionAndFees()# - #getFallOrientationFee()#
	<br />#calGrantA#
	<br />#calGrantB#
	<br />#lsuAllowance#
	<br />#sdaAward#
	<br />#lsuPerformance#
	<br />#familyDiscount#
	<br />#nationalMerit#
	<br />#churchMatch#
	<br />#pacificCampMatch#
	<br />#nonPacificCampMatch#
	<br />#litEvanMatch#
</CFOUTPUT
**********/ 
	 _lsuSubtotal = round(calGrantA
			    + calGrantB
			    + lsuAllowance
	                    + sdaAward
	                    + lsuPerformance
	                    + familyDiscount
	                    + nationalMerit
	                    + churchMatch
	                    + pacificCampMatch
	                    + nonPacificCampMatch
	                    + litEvanMatch)

	if(  calGrantA  > 0
           or calGrantB  > 1551
		 _lsuLimit = getTuitionAndFees() - getFallOrientationFee() + 1000
	}else{
		 _lsuLimit = getTuitionAndFees() - getFallOrientationFee()
	}

	 _lsuLimitSubtotal = calGrantA
	if(  _lsuLimitSubtotal  > _lsuLimit
		 calGrantA           = _lsuLimit
		 calGrantB           = 0
		 lsuAllowance        = 0
		 sdaAward            = 0
		 lsuPerformance      = 0
		 familyDiscount      = 0
		 nationalMerit       = 0
		 churchMatch         = 0
		 pacificCampMatch    = 0
		 nonPacificCampMatch = 0
		 litEvanMatch        = 0
		return  _lsuSubtotal
	}

	 _lsuLimitSubtotal = _lsuLimitSubtotal + calGrantB
	if(  _lsuLimitSubtotal  > _lsuLimit
		 calGrantB           = calGrantB - (_lsuLimitSubtotal - _lsuLimit)
		 lsuAllowance        = 0
		 sdaAward            = 0
		 lsuPerformance      = 0
		 familyDiscount      = 0
		 nationalMerit       = 0
		 churchMatch         = 0
		 pacificCampMatch    = 0
		 nonPacificCampMatch = 0
		 litEvanMatch        = 0
		return  _lsuSubtotal
	}

	 _lsuLimitSubtotal = _lsuLimitSubtotal + lsuAllowance
	if(  _lsuLimitSubtotal  > _lsuLimit
		 lsuAllowance        = lsuAllowance - (_lsuLimitSubtotal - _lsuLimit)
		 sdaAward            = 0
		 lsuPerformance      = 0
		 familyDiscount      = 0
		 nationalMerit       = 0
		 churchMatch         = 0
		 pacificCampMatch    = 0
		 nonPacificCampMatch = 0
		 litEvanMatch        = 0
		return  _lsuSubtotal
	}

	 _lsuLimitSubtotal = _lsuLimitSubtotal + sdaAward
	if(  _lsuLimitSubtotal  > _lsuLimit
		 sdaAward            = sdaAward - (_lsuLimitSubtotal - _lsuLimit)
		 lsuPerformance      = 0
		 familyDiscount      = 0
